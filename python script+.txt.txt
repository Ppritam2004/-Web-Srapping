    "import requests\n",
    "from bs4 import BeautifulSoup\n",
    "\n",
    "def fetch_headlines(url=\"https://www.indiatoday.in/\"):\n",
    "    headers = {\n",
    "        \"User-Agent\": \"Mozilla/5.0 (Windows NT 10.0; Win64; x64)\"\n",
    "    }\n",
    "    resp = requests.get(url, headers=headers)\n",
    "    resp.raise_for_status()\n",
    "    soup = BeautifulSoup(resp.text, \"html.parser\")\n",
    "\n",
    "    headlines = set()\n",
    "\n",
    "    # Extract <h1>, <h2>, <h3> tags\n",
    "    for tag in soup.find_all(['h1', 'h2', 'h3']):\n",
    "        text = tag.get_text(strip=True)\n",
    "        if len(text) > 10:\n",
    "            headlines.add(text)\n",
    "\n",
    "    # Extract links that match article pattern\n",
    "    for a in soup.select(\"a[href*='/story/']\"):\n",
    "        text = a.get_text(strip=True)\n",
    "        if 15 <= len(text) <= 100:\n",
    "            headlines.add(text)\n",
    "\n",
    "    return list(headlines)\n",
    "\n",
    "def save_headlines(headlines, filename=\"india_today_headlines.txt\"):\n",
    "    with open(filename, 'w', encoding='utf-8') as f:\n",
